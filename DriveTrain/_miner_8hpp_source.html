173 if (_miningStartTime == 0)
174 _miningStartTime = _now;
175
176 // Safety: if mining indefinitely but exceeded configured maximum continuous time, stop.
177 if (_hits_to_mine == INDEFINITE && _maxContinuousMillis > 0 &&

179 {
180 // mark timed out and stop

183 Serial.println("Miner timed out (max continuous time exceeded)");
184 servo.update();
185 return;
186 }

188 const unsigned long cycleMs = MINER_CYCLE_MS;
189 const unsigned long pressMs = MINER_PRESS_MS;
190
191 // Start a new cycle if needed
192 if (_cycleStartTime == 0)
193 {
194 _cycleStartTime = _now;
195 _onStartTime = _now; // start pressing immediately at cycle start
196 setServoToPress();
197 servo.update();
198 return;
199 }
200
201 // How long since the cycleStart (mod cycle length) — keeps values bounded
202 unsigned long elapsedSinceStart = _now - _cycleStartTime;
203 unsigned long elapsedInCycle = elapsedSinceStart % cycleMs;
204
205 // If within press window -> ensure pressing
206 if (elapsedInCycle < pressMs)
207 {
208 // if we just entered press window (no onStart recorded or long past), record
209 if (_onStartTime == 0 || (_now - _onStartTime) > cycleMs)
210 {
211 _onStartTime = _now;
212 setServoToPress();
213 }
214 // otherwise keep pressing (no-op)
215 }
216 else
217 {
218 // outside press window -> ensure retracted and reset onStart
219 _onStartTime = 0;
220 setServoToRetract();
221
222 // If one or more full cycles have elapsed since _cycleStartTime, advance and count hits
223 if (elapsedSinceStart >= cycleMs)
224 {
225 // How many cycles have passed fully since _cycleStartTime
226 unsigned long cyclesPassed = elapsedSinceStart / cycleMs;
227
228 // Prevent extremely large jumps (protect against very long pauses)
229 const unsigned long MAX_CYCLES_INCREMENT = 10000UL;
230 if (cyclesPassed > MAX_CYCLES_INCREMENT)
231 cyclesPassed = MAX_CYCLES_INCREMENT;
232
233 _cycleStartTime += cyclesPassed * cycleMs;
234
235 // Increase hit count by the number of completed cycles
236 _number_hits += static_cast<uint32_t>(cyclesPassed);
237
238 // If we reached / exceeded a finite goal, clamp and stop
239 if (_hits_to_mine != INDEFINITE && _number_hits >= static_cast<uint32_t>(_hits_to_mine))

243 }
244 }
