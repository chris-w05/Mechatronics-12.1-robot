110 IndexDirection whereToIndexBlock(Block newBlock){
111 //Fill wood, biasing left
112 if( _nextItem == PICKAXE_HANDLE){
113 if( newBlock == WOOD){
114 //If there is already a block there, HOLD, otherwise, push it left

121 //Left, Right, center order.

139 //if new block is wood

141 //Fill to left if empty

146
147 //Blocks in this part are guaranteed to not be wood
148 //Take whatever possible - this could be changed later depending on block frequency
149 if( _possession[RIGHT] == NONE) return RIGHT;
