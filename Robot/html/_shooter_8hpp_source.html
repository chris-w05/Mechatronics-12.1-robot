264 void handleAutoFire()
265 {
266 unsigned long now = millis();
267 switch( autoFire_state){
268 case WAITFORBLOCK:
269 {
270 // Detect rising edge of switch
271 block_switch.update();
272 bool reading = block_switch.getReading();
273 if (reading && reading != last_switch_value)
274 {
275 _blockDetectedTime = now;
276 autoFire_state = HASBLOCK;
277 }
278 break;
279 }
280
281 case HASBLOCK:
282 {
283 if( now - _blockDetectedTime > SHOOTER_SETTLE_TIME){
284 //After enough time has passed for block to settle, fire the shooter
285 targetPosition = nextFirePosition();
286 motor.setTarget(targetPosition);
287 autoFire_state = FIRE;
288 _fireTime = now;
289 }
290 break;
291 }
292
293 case FIRE:
294 {
295 //Allow rack to settle before firing again
296 if( now - _fireTime > SHOOTER_FIRE_TIME ){
297 targetPosition = nextPrimePosition();
298 motor.setTarget(targetPosition);
299 //Set shooter to wait for block state;
300 autoFire_state = WAITFORBLOCK;
301 }
302 break;
303 }
304
305 }
306 //Update motor controller regardless of state
307 motor.update(position, velocity);
308 }
