217 void handleAutoFire()
218 {
219 unsigned long now = millis();
220 switch( autoFire_state){
221 case WAITFORBLOCK:
222 {
223 // Detect rising edge of switch
224 block_switch.update();
225 bool reading = block_switch.getReading();
226 if (reading && reading != last_switch_value)
227 {
228 _blockDetectedTime = now;
229 autoFire_state = HASBLOCK;
230 }
231 break;
232 }
233
234 case HASBLOCK:
235 {
236 if( now - _blockDetectedTime > SHOOTER_SETTLE_TIME){
237 //After enough time has passed for block to settle, fire the shooter
238 targetPosition = nextFirePosition();
239 motor.setTarget(targetPosition);
240 autoFire_state = FIRE;
241 _fireTime = now;
242 }
243 break;
244 }
245
246 case FIRE:

248 //Allow rack to settle before firing again

252 //Set shooter to wait for block state;
253 autoFire_state = WAITFORBLOCK;

258 }
