135 virtual float update(float measurement, float dmeasurement, float setpoint)
136 {
137 unsigned long now = micros();
138 unsigned long dt_ms = now - _lastTime;
139
140 float dt = dt_ms * 0.000001; // seconds
141
142 float error = setpoint - measurement;
143
144 // Integrator
145 _integral += error * dt;
146
147 _derivative = dmeasurement;
148
149 float output =
150 _kp * error - _kd * _derivative + _ki * _integral;
151
152 // Save state
153 _lastMeasurement = measurement;
154 _lastTime = now;
155 _lastOutput = output;
156

345 float update(float measurement, float dmeasurement, float setpoint, float dSetpoint = 0)
346 {
347 unsigned long now = micros();
348 unsigned long dt_micros = now - _lastTime;
349
350 if (dt_micros == 0)
351 return _lastOutput;
352
353 float dt = dt_micros * 0.000001; // seconds
354
355 float error = setpoint - measurement;
356
357 // Integrator
358 _integral += error * dt;
359 _integral = constrain(_integral, _iMin, _iMax); // anti-windup via clamping
360
361 float output = calculate(measurement, setpoint, dmeasurement, _integral, dSetpoint);
362
363 // Save state
364 _lastMeasurement = measurement;
365 _lastTime = now;
366 _lastOutput = output;

376 float update(float measurement, float setpoint)
377 {
378 unsigned long now = micros();
379 unsigned long dt_micross = now - _lastTime;
380
381 if (dt_micross == 0)
382 return _lastOutput;
383
384 float dt = dt_micross * 0.000001; // seconds
385
386 float error = setpoint - measurement;
387
388 // Integrator
389 _integral += error * dt;
390 _integral = constrain(_integral, _iMin, _iMax); // anti-windup via clamping
391
392 // Derivative (on measurement to avoid derivative kick from setpoint)
393 float derivative = (measurement - _lastMeasurement) / dt;
394 derivative = (.1)*derivative + (.9)*_lastDerivative; //IIR filter with alpha = .1
395
396
397 // Feedforward: if user supplied a custom function use it; otherwise use simple kff * setpoint
398 float output = calculate(measurement, setpoint, derivative, _integral, 0);
399
400 // Save state
401 _lastMeasurement = measurement;
402 _lastTime = now;
403 _lastOutput = output;
404 _lastDerivative = derivative;
405
406
407
408 return output;

488 float calculate(float measurement, float setpoint, float derivative, float integral, float dSetpoint){
489 float error = setpoint - measurement;
490 float feedforward = 0.0;
491
492 // nonlinear optional functions. In most cases these are zero.
493 if (_ffFunc)
494 {
495 feedforward = _ffFunc(setpoint);
496 }
497 else
498 {
499 feedforward = _kff * setpoint;
500 }
501
502 float dFeedForward = 0.0;
503 if (_dffFunc){
504 dFeedForward = _dffFunc(dSetpoint);
505 }
506
507 float kpNonlinear = 0.0;
508 if (_kpFunc)
509 {
510 kpNonlinear = _kpFunc(measurement, setpoint);
511 }
512
513 float kiNonlinear = 0.0;
514 if (_kiFunc)
515 {
516 kiNonlinear = _kiFunc(_integral);
517 }
518
519 float kdNonlinear = 0.0;
520 if (_kiFunc)
521 {
522 kdNonlinear = _kdFunc(derivative);
523 }
524 float signal = _kp * error - _kd * (derivative - dSetpoint) + _ki * integral +
525 feedforward + dFeedForward + kpNonlinear + kiNonlinear + kdNonlinear;
526
527 signal = (_alpha * signal) + (1-_alpha)*_lastOutput;
528 _lastOutput = signal;
529 return signal;
530
