129 void update() override
130 {
131 unsigned long now = micros();
132 float dt = (now - _lastTime) * 1e-6f;
133 _lastTime = now;
134
135 // --- Update sensors ---
136 // Encoders are always needed for odometry and PID.
137 _leftEncoder.update();
138 _rightEncoder.update();
139 // Gate expensive peripheral reads to the modes that actually use them.
140 // The QTR RC line sensor blocks for up to its timeout (default 1000 µs) per read;
141 // the distance sensor adds ~100 µs for analogRead().
142 if (_mode == LINEFOLLOWING || _mode == LINEFOLLOWING_HARDSET)
143 _lineSensor.update();
144 if (_mode == DISTANCE)
145 _distSensor.update();
146
147 float leftPos = _leftEncoder.getCount() * DRIVETRAIN_TICKS_TO_IN;
148 float rightPos = _rightEncoder.getCount() * DRIVETRAIN_TICKS_TO_IN;
149 float leftVel = _leftEncoder.getVelocity() * DRIVETRAIN_TICKS_TO_IN; // in/s
150 float rightVel = _rightEncoder.getVelocity()* DRIVETRAIN_TICKS_TO_IN;
151
152 // --- Mode-specific control ---
153 switch (_mode) {
154
155 case LINEFOLLOWING: {
156 float correction = _lineSensor.getPosition()
157 * DRIVE_LINEFOLLOW_VELOCITY_GAIN * _speedL;
158 float newSpeedL = _speedL + correction / LINESENSOR_LR_RATIO;
159 float newSpeedR = _speedR - correction;
160
161 _leftTargetPos += newSpeedL * dt;
162 _rightTargetPos += newSpeedR * dt;
163
164 _motorController.setTarget(_leftTargetPos, _rightTargetPos);
165 _motorController.update(leftPos, leftVel, rightPos, rightVel,
166 _speedL, _speedR);
167 break;
168 }
169
170 case STRAIGHT:
171 case ARC:
172 case STOPPED: {
173 // Ramp the actual setpoint toward the target to limit acceleration
174 const float maxDelta = 20.0f * dt;
175 float deltaL = _targetSpeedL - _speedL;
176 float deltaR = _targetSpeedR - _speedR;
177 float maxChange = max(fabsf(deltaL), fabsf(deltaR));
178
179 if (maxChange > maxDelta) {
180 float scale = maxDelta / maxChange;
181 _speedL += deltaL * scale;
182 _speedR += deltaR * scale;
183 } else {
184 _speedL = _targetSpeedL;
185 _speedR = _targetSpeedR;
186 }
187
188 // Clamp feedforward wheel speeds
189 float vL_ref = constrain(_speedL, -MAXVELOCITY, MAXVELOCITY);
190 float vR_ref = constrain(_speedR, -MAXVELOCITY, MAXVELOCITY);
191
192 // Feedforward chassis velocities
193 float vRef = 0.5f * (vL_ref + vR_ref);
194 float wRef = (vR_ref - vL_ref) / DRIVETRAIN_WIDTH;
195
196 // Ramsete correction toward the feedforward pose
197 float vCmd = vRef, wCmd = wRef;
198 if (_ramsete.isEnabled()) {
199 RamseteController::Cmd cmd = _ramsete.step(
200 _desiredOdometry.getPose(), vRef, wRef,
201 _odometry.getPose());
202 vCmd = cmd.v;
203 wCmd = cmd.w;
204 }
205
206 // Convert chassis → individual wheel speeds
207 float halfW = DRIVETRAIN_WIDTH * 0.5f;
208 float vL_cmd = constrain(vCmd - halfW * wCmd, -MAXVELOCITY, MAXVELOCITY);
209 float vR_cmd = constrain(vCmd + halfW * wCmd, -MAXVELOCITY, MAXVELOCITY);
210
211 _leftTargetPos += vL_cmd * dt;
212 _rightTargetPos += vR_cmd * dt;
213
214 _motorController.setTarget(_leftTargetPos, _rightTargetPos);
215 _motorController.update(leftPos, leftVel, rightPos, rightVel,
216 vL_cmd, vR_cmd);
217
218 // Emit telemetry at 20 Hz for real-time plotting / debugging
219 if (millis() - _lastTelemetryMs >= 50) {
220 _lastTelemetryMs = millis();
221 emitTelemetryJSON(
222 Serial,
223 _odometry, _desiredOdometry,
224 leftVel, rightVel,
225 vL_cmd, vR_cmd,
226 leftPos, rightPos,
227 _leftTargetPos, _rightTargetPos);
228 }
229
230 // Advance the feedforward (ideal) trajectory
231 _desiredOdometry.update(_targetSpeedL * dt, _targetSpeedR * dt);
232 break;
233 }
234
235 case LINEFOLLOWING_HARDSET: {
236 float correction = _lineSensor.getPosition();
237 Serial.println(correction);
238 correction *= DRIVE_LINEFOLLOW_GAIN * (_signalL + _signalR) / 2;
239 _motorController.setPower(
240 _signalL + (int)(correction / LINESENSOR_LR_RATIO),
241 _signalR - (int)correction);
242 break;
243 }
244
245 case HARDSET:
246 _motorController.setPower(_signalL, _signalR);
247 break;
248
249 case DISTANCE: {
250 float dist = _distSensor.getDistanceCm();
251 _motorController.setTarget(_targetDistance, _targetDistance);
252 _motorController.updateWithoutDerivatice(dist, dist);
253 break;
254 }
255
256 default:
257 _motorController.setPower(0, 0);
258 }
259
260 // Update true-pose odometry from encoder ticks
261 _odometry.update(_leftEncoder, _rightEncoder);
262 }

335 void followRadiusAtVelocity(float velocity, float radius, bool resetPID = true)
336 {
337 _mode = MODE::ARC;
338 if (resetPID) _motorController.resetPID();
339 if (_resetTargetPosBetweenSegments){
340 syncTargetPositions();
341 }
342
343 const float halfL = DRIVETRAIN_WIDTH / 2.0f;
344
345 if (fabsf(radius) < 1e-6f) {
346 // Spin in place: positive velocity → clockwise
347 _targetSpeedL = -velocity;
348 _targetSpeedR = velocity;
349 return;
350 }
351
352 float omega = -velocity / radius; // sign of radius determines turn direction
353 _targetSpeedL = velocity - halfL * omega;
354 _targetSpeedR = velocity + halfL * omega;
355 }
