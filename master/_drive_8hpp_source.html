148 void update() override
149 {
150 unsigned long now = micros();
151 float dt = (now - _lastTime) * 1e-6f;
152 _lastTime = now;
153
154 // --- Update sensors ---
155 // Encoders are always needed for odometry and PID.
156 _leftEncoder.update();
157 _rightEncoder.update();
158 // Gate expensive peripheral reads to the modes that actually use them.
159 // The QTR RC line sensor blocks for up to its timeout (default 1000 µs) per read;
160 // the distance sensor adds ~100 µs for analogRead().
161 if (_mode == LINEFOLLOWING || _mode == LINEFOLLOWING_HARDSET || _mode == LINEFOLLOWING_DISTANCE)
162 _lineSensor.update();
163 if (_mode == DISTANCE || _mode == LINEFOLLOWING_DISTANCE)
164 _distSensor.update();
165
166 float leftPos = _leftEncoder.getCount() * DRIVETRAIN_TICKS_TO_IN;
167 float rightPos = _rightEncoder.getCount() * DRIVETRAIN_TICKS_TO_IN;
168 float leftVel = _leftEncoder.getVelocity() * DRIVETRAIN_TICKS_TO_IN; // in/s
169 float rightVel = _rightEncoder.getVelocity()* DRIVETRAIN_TICKS_TO_IN;
170
171 // --- Mode-specific control ---
172 switch (_mode) {
173
174 case LINEFOLLOWING: {
175 float correction = _lineSensor.getPosition()
176 * DRIVE_LINEFOLLOW_VELOCITY_GAIN * _speedL;
177 float newSpeedL = _speedL + correction / LINESENSOR_LR_RATIO;
178 float newSpeedR = _speedR - correction;
179
180 _leftTargetPos += newSpeedL * dt;
181 _rightTargetPos += newSpeedR * dt;
182
183 _motorController.setTarget(_leftTargetPos, _rightTargetPos);
184 _motorController.update(leftPos, leftVel, rightPos, rightVel,
185 _speedL, _speedR);
186 break;
187 }
188
189 case STRAIGHT:
190 case ARC:
191 case STOPPED: {
192 // Ramp the actual setpoint toward the target to limit acceleration
193 const float maxDelta = 3000.0f * dt;
194 float deltaL = _targetSpeedL - _speedL;
195 float deltaR = _targetSpeedR - _speedR;
196 float maxChange = max(fabsf(deltaL), fabsf(deltaR));
197
198 if (maxChange > maxDelta) {
199 float scale = maxDelta / maxChange;
200 _speedL += deltaL * scale;
201 _speedR += deltaR * scale;
202 } else {
203 _speedL = _targetSpeedL;
204 _speedR = _targetSpeedR;
205 }
206
207 // Clamp feedforward wheel speeds
208 float vL_ref = constrain(_speedL, -MAXVELOCITY, MAXVELOCITY);
209 float vR_ref = constrain(_speedR, -MAXVELOCITY, MAXVELOCITY);
210
211 // Feedforward chassis velocities
212 float vRef = 0.5f * (vL_ref + vR_ref);
213 float wRef = (vR_ref - vL_ref) / DRIVETRAIN_WIDTH;
214
215 // Ramsete correction toward the feedforward pose
216 float vCmd = vRef, wCmd = wRef;
217 if (_ramsete.isEnabled()) {
218 RamseteController::Cmd cmd = _ramsete.step(
219 _desiredOdometry.getPose(), vRef, wRef,
220 _odometry.getPose());
221 vCmd = cmd.v;
222 wCmd = cmd.w;
223 }
224
225 // Convert chassis → individual wheel speeds
226 float halfW = DRIVETRAIN_WIDTH * 0.5f;
227 float vL_cmd = constrain(vCmd - halfW * wCmd, -MAXVELOCITY, MAXVELOCITY);
228 float vR_cmd = constrain(vCmd + halfW * wCmd, -MAXVELOCITY, MAXVELOCITY);
229
230 _leftTargetPos += vL_cmd * dt;
231 _rightTargetPos += vR_cmd * dt;
232
233 _motorController.setTarget(_leftTargetPos, _rightTargetPos);
234 _motorController.update(leftPos, leftVel, rightPos, rightVel,
235 vL_cmd, vR_cmd);
236
237 // Emit telemetry at 20 Hz for real-time plotting / debugging
238 // if (millis() - _lastTelemetryMs >= 50) {
239 // _lastTelemetryMs = millis();
240 // emitTelemetryJSON(
241 // Serial,
242 // _odometry, _desiredOdometry,
243 // leftVel, rightVel,
244 // vL_cmd, vR_cmd,
245 // leftPos, rightPos,
246 // _leftTargetPos, _rightTargetPos);
247 // }
248
249 // Advance the feedforward (ideal) trajectory
250 _desiredOdometry.update(_targetSpeedL * dt, _targetSpeedR * dt);
251 break;
252 }
253
254 case LINEFOLLOWING_HARDSET: {
255 float correction = _lineSensor.getPosition();
256 float correction_signal = _lineSensorPID.update(correction, 0); // Scale signal by how fast the robot is moving
257
258 _motorController.setPower(
259 _signalL + (int)(correction_signal / LINESENSOR_LR_RATIO), // Accounts for offcenter position of line sensor
260 _signalR - (int)correction_signal);
261 break;
262 }
263
264 case LINEFOLLOWING_DISTANCE:{
265 //Distance sensor sets the base signal for control:
266 float dist = _distSensor.getDistanceIn();
267 float signalBase = _distanceSensorPID.update(dist, _targetDistance);
268 _signalL = signalBase;
269 _signalR = signalBase;
270 //Line sensor applies steering correction to distance sensor commands
271 float correction = _lineSensor.getPosition();
272 float correction_signal = _lineSensorPID.update(correction, 0); //Scale signal by how fast the robot is moving
273 _motorController.setPower(
274 _signalL + (int)(correction_signal / LINESENSOR_LR_RATIO), //Accounts for offcenter position of line sensor
275 _signalR - (int)correction_signal);
276 break;
277 }
278
279 case HARDSET:
280 _motorController.setPower(_signalL, _signalR);
281 break;
282
283 case DISTANCE: {
284 // Distance sensor sets the base signal for control:
285 float dist = _distSensor.getDistanceIn();
286 float signalBase = _distanceSensorPID.update(dist, _targetDistance);
287 _signalL = signalBase;
288 _signalR = signalBase;
289 _motorController.setPower(
290 _signalL, // Accounts for offcenter position of line sensor
291 _signalR);
292 break;
293 }
294
295 default:
296 _motorController.setPower(0, 0);
297 }
298
299 // Update true-pose odometry from encoder ticks
300 _odometry.update(_leftEncoder, _rightEncoder);
301 }

354 void followRadiusClockwise(float omega_rad_s, float radius)
355 {
356 _mode = MODE::ARC;
357 _motorController.makeLinear();
358 if (_resetTargetPosBetweenSegments)
359 {
360 syncTargetPositions();
361 }
362 const float halfL = DRIVETRAIN_WIDTH / 2.0f;
363 _targetSpeedL = omega_rad_s * (radius + halfL);
364 _targetSpeedR = omega_rad_s * (radius - halfL);
365 }
366
374 void followRadiusAtVelocity(float velocity, float radius, bool resetPID = true)
375 {
376 _mode = MODE::ARC;
377 if (resetPID) _motorController.resetPID();
378 if (_resetTargetPosBetweenSegments){
379 syncTargetPositions();
380 }
381
382 const float halfL = DRIVETRAIN_WIDTH / 2.0f;
383
384 if (fabsf(radius) < 1e-6f) {
385 // Spin in place: positive velocity → clockwise
386 _targetSpeedL = -velocity;
387 _targetSpeedR = velocity;
388 return;
389 }
390
391 float omega = -velocity / radius; // sign of radius determines turn direction
392 _targetSpeedL = velocity - halfL * omega;
393 _targetSpeedR = velocity + halfL * omega;
394 }
