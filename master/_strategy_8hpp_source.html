171 IndexDirection whereToIndexBlock(Block newBlock){
172 //Fill wood, biasing left
173 if( _nextItem == PICKAXE_HANDLE){
174 if( newBlock == WOOD){
175 //If there is already a block there, HOLD, otherwise, push it left
176 if (_possession[LEFT] != NONE) return HOLD;
177 else return LEFT;
178 }
179 return REJECT;
180 }
181
182 //Left, Right, center order.
183 //Also needs to make sure all blocks are same type
184 if (_nextItem == PICKAXE_TIP){
185 if( newBlock != getMaxMineableBlock()){
186 //Potentially only chest valuable blocks - woudl require a more thourough analysis
187 return CHEST;
188 }
189 //All blocks owned will be max level
190
191 //Put block in LEFT if empty, same with RIGHT
192 if( _possession[LEFT] == NONE) return LEFT;
193 if (_possession[RIGHT] == NONE) return RIGHT;
194 //HOLD if left and right are filled
195 return IndexDirection::HOLD;
196 }
197
198
199 if (_nextItem == SWORD){
200 //if new block is wood
201 if (newBlock == WOOD){
202 //Fill to left if empty
203 if( _possession[LEFT] == NONE) return LEFT;
204 //otherwise get rid of the block
205 return CHEST;
206 }
207
208 //Blocks in this part are guaranteed to not be wood
209 //Take whatever possible - this could be changed later depending on block frequency
210 if( _possession[RIGHT] == NONE) return RIGHT;
211 if( newBlock != _possession[RIGHT]) return REJECT;
212 return HOLD;
213 }
214
215 return REJECT;
216 }
