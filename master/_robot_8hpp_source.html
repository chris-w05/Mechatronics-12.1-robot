100 void update()
101 {
102 // Read both command channels
103 serialComs.update();
104 updateUsbSerial();
105
106 if (serialComs.hasCommand())
107 processCommandString(serialComs.getCommand(), &Serial2);
108
109 if (_usbHasCommand) {
110 _usbHasCommand = false;
111 processCommandString(_usbBuffer, &Serial);
112 }
113
114 // Tick all subsystems every loop
115 for (int i = 0; i < subsystemCount; ++i)
116 subsystems[i]->update();
117
118 // Tick the autonomous scheduler when active
119 if (mode == AUTONOMOUS || mode == SERIAL_TEST)
120 autonomous.update();
121 }
