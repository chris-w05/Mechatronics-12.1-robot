155 {
156 if (!cmdStr || cmdStr[0] == '\0')
157 return false;
158
159 // Copy into modifiable buffer and normalize separators to spaces
160 char buf[SerialComs::MAX_CMD_LEN];
161 strncpy(buf, cmdStr, sizeof(buf));
162 buf[sizeof(buf) - 1] = '\0';
163 for (char *p = buf; *p; ++p)
164 if (*p == ',' || *p == ':')
165 *p = ' ';
