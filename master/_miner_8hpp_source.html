72 void update() override
73 {
74 _now = millis();
75
76 // If programmed to stop after a fixed number of hits, and we've reached that,
77 // stop mining. Note: compare only if _hits_to_mine is not INDEFINITE.
78 if (_hits_to_mine != INDEFINITE && _number_hits >= static_cast<uint32_t>(_hits_to_mine))
79 {
80 stopMiningInternal();
81 }
82
83 // If we timed out previously, keep miner OFF until explicitly restarted
84 if (_timedOut)
85 {
86 // ensure minerServo retracted and don't auto-reset timers
87 setMinerServoToRetract();
88 minerServo.update();
89 return;
90 }
91
92 switch(_mode){
93
94 case LIFT:
95 setRampServoToLift();
96 rampServo.update();
97 return;
98
99 case STORE:
100 // Ensure minerServo is retracted while stopped and timers cleared for next start
101 setServosToStore();
102 _cycleStartTime = 0;
103 _onStartTime = 0;
104 minerServo.update();
105 rampServo.update();
106 return;
107
108 case OFF:
109 // Ensure minerServo is retracted while stopped and timers cleared for next start
110 setMinerServoToRetract();
111 _cycleStartTime = 0;
112 _onStartTime = 0;
113 minerServo.update();
114 return;
115
116 case MINING:
117 mine();
118 setRampServoToPassive();
119 rampServo.update();
120 return;

218 void mine(){
219 // MINING mode ---------------------------------------------------------
220 // Start global mining timer on first transition to MINING
221 if (_miningStartTime == 0)
222 _miningStartTime = _now;
223
224 // Safety: if mining indefinitely but exceeded configured maximum continuous time, stop.
225 if (_hits_to_mine == INDEFINITE && _maxContinuousMillis > 0 &&
226 (_now - _miningStartTime) >= _maxContinuousMillis)
227 {
228 // mark timed out and stop
229 _timedOut = true;
230 stopMiningInternal();
231 Serial.println("Miner timed out (max continuous time exceeded)");
232 minerServo.update();
233 return;
234 }
235
236 const unsigned long cycleMs = MINER_CYCLE_MS;
237 const unsigned long pressMs = MINER_PRESS_MS;
238
239 // Start a new cycle if needed
240 if (_cycleStartTime == 0)
241 {
242 _cycleStartTime = _now;
243 _onStartTime = _now; // start pressing immediately at cycle start
244 setMinerServoToPress();
245 minerServo.update();
246 return;
247 }
248
249 // How long since the cycleStart (mod cycle length) — keeps values bounded
250 unsigned long elapsedSinceStart = _now - _cycleStartTime;
251 unsigned long elapsedInCycle = elapsedSinceStart % cycleMs;
252
253 // If within press window -> ensure pressing
254 if (elapsedInCycle < pressMs)
255 {
256 // if we just entered press window (no onStart recorded or long past), record
257 if (_onStartTime == 0 || (_now - _onStartTime) > cycleMs)
258 {
259 _onStartTime = _now;
260 setMinerServoToPress();
261 }
262 // otherwise keep pressing (no-op)
263 }
264 else
265 {
266 // outside press window -> ensure retracted and reset onStart
267 _onStartTime = 0;
268 setMinerServoToRetract();
269
270 // If one or more full cycles have elapsed since _cycleStartTime, advance and count hits
271 if (elapsedSinceStart >= cycleMs)
272 {
273 // How many cycles have passed fully since _cycleStartTime
274 unsigned long cyclesPassed = elapsedSinceStart / cycleMs;
275
276 // Prevent extremely large jumps (protect against very long pauses)
277 const unsigned long MAX_CYCLES_INCREMENT = 10000UL;
278 if (cyclesPassed > MAX_CYCLES_INCREMENT)
279 cyclesPassed = MAX_CYCLES_INCREMENT;
280
281 _cycleStartTime += cyclesPassed * cycleMs;
282
283 // Increase hit count by the number of completed cycles
284 _number_hits += static_cast<uint32_t>(cyclesPassed);
285// if ((_number_hits % 10) == 0)
286// {
287// setServoToIdle();
288// rampServo.update();
289// }
290
291 // If we reached / exceeded a finite goal, clamp and stop
292 if (_hits_to_mine != INDEFINITE && _number_hits >= static_cast<uint32_t>(_hits_to_mine))
293 {
294 stopMiningInternal();
295 }
296 }
297 }
298
299 // Always update minerServo at end so PWM/system-level updates occur
300 minerServo.update();
301 }
